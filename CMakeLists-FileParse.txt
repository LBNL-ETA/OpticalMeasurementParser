cmake_minimum_required( VERSION 2.8.7 )

# Create include directories that will connect to library itself. This must be created before downloading 
# FileParse library or otherwise connections will not be created

set(FileParse_INCLUDE_DIRS "${CMAKE_BINARY_DIR}/FileParse-src/include")

include_directories(${FileParse_INCLUDE_DIRS})

# Download and create Windows Calculation Engine library
configure_file(CMakeLists-FileParse.txt.in ${CMAKE_BINARY_DIR}/FileParse-download/CMakeLists.txt)
execute_process(COMMAND ${CMAKE_COMMAND} -G "${CMAKE_GENERATOR}" .
  WORKING_DIRECTORY ${CMAKE_BINARY_DIR}/FileParse-download)
execute_process(COMMAND ${CMAKE_COMMAND} --build .
  WORKING_DIRECTORY ${CMAKE_BINARY_DIR}/FileParse-download)
 
add_subdirectory(${CMAKE_BINARY_DIR}/FileParse-src "${CMAKE_CURRENT_BINARY_DIR}/FileParse-src")
set(FileParse_LIB "${CMAKE_SHARED_LIBRARY_PREFIX}FileParse${CMAKE_SHARED_LIBRARY_SUFFIX}")
